float4 main(PixelShaderInput input) : SV_TARGET {
    float x = input.texcoord.x;
    float y = input.texcoord.y;
    
    float f = (x - 0.5 - xSplit) * (x - 0.5 - xSplit) +
             (y - 0.5) * (y - 0.5) -
             0.3 * 0.3;

    float4 color = g_texture0.Sample(g_sampler, input.texcoord);

    if (f < 0.0)
       return color * 1.5;
    else
       return color * 0.3;
}
